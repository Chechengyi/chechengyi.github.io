# sequelize使用记录

# 建立多对多关系

场景：在做权限控制的时候，一般有一张用户 user 表和一张角色 role 表， 用户和角色之间的关系是多对多的，通过一张用户角色表建立起关系。

用户表的定义：

const User = sequelize.define('users', {
    id: {
        type: Sequelize.STRING(100),
        primaryKey: true,
        defaultValue: uuid.v1()
    },
    sex: {
        type: Sequelize.INTEGER
    },
    username: {
        type: Sequelize.STRING(100),
        allowNull: false
    },
    password: {
        type: Sequelize.STRING(100),
        allowNull: false
    }
}, {
    freezeTableName: false,
    timestamps: true,
    paranoid: true
});

角色表的定义：

const Role = sequelize.define('role', {
  id: {
    type: Sequelize.INTEGER,
    primaryKey: true,
    autoIncrement: true
  },
  roleName: {
    type: Sequelize.STRING,
    allowNull: false
  },
  desc: Sequelize.STRING
}, {
});

接下来中间表 用户角色表的定义：

const UserRole = sequelize.define('user_role', {
  id: {
    type: Sequelize.INTEGER,
    primaryKey: true,
    autoIncrement: true
  },
  user_id: Sequelize.STRING,
  role_id: Sequelize.INTEGER
}, {
  timestamps: false
});

export default UserRole;

# 建立外键关系

User.belongsToMany(Role, {
    as: 'roles',
    through: UserRole,
    foreignKey: 'user_id'
});
Role.belongsToMany(User, {
    as: 'users',
    through: UserRole,
    foreignKey: 'role_id'
})

# 查询使用

let res = await Users.findAll({
      include: [{
        model: model.Role,
        as: 'roles'
      }],
      where: {
        username,
        password

    })
